## Stories

The CSF rendered inside the `Canvas` from which you can see the implementation example.

### Default

<Canvas>
	<Story name='Default'>
		{() => {
			const key = 'something';
			const search = true;
			const labelSearch = true;
			const valueSearch = true;
			const autocompleteOptions = {
				render: ({ value }) => {
					return value;
				},
			};
			const [label, setLabel] = useState(null);
			const [value, setValue] = useState(null);
			return (
				<ThemedContainer
					key={key}
					search={true}
					labelSearch={true}
					valueSearch={true}
					autocompleteOptions={autocompleteOptions}>
					<TableChipItem
						label={label}
						value={value}
						key={key}
						search={true}
						labelSearch={true}
						valueSearch={true}
						autocompleteOptions={autocompleteOptions}
						onSearch={(event) => {
							const { fieldValue, dataset } = inputHelper(event);
							const { search } = dataset;
							if (search === 'label') {
								setLabel(fieldValue);
							} else if (search === 'value') {
								setValue(fieldValue);
							}
						}}
					/>
				</ThemedContainer>
			);
		}}
	</Story>
</Canvas>
