import { Meta, Canvas, Story } from '@storybook/addon-docs';
import { useState } from 'react';

import TableRow from './TableRow';
import { Button } from '../../buttons';
import { TableCellV2 } from '../cell';
import Text from '../../text/Text';

<Meta title='Components/v2/Table/Row' parameters={{ options: { showToolbar: false } }} />

export const Template = (args) => {
	const [checkedRows, setCheckedRows] = useState([]);
	return <TableRow {...args} checkedRows={checkedRows} setCheckedRows={setCheckedRows} />;
};

# Table Row

> Container for table rows of table body.

## Usage

```js
import { TableRow } from '@banyan_cloud/roots';
```

### Expandable

<Canvas>
	<Story
		name='Expandable'
		args={{
			datum: {
				id: 2,
				name: 'Aryan Dixit',
				age: 27,
				designation: 'UI Engineer',
				state: 'UP',
			},
			uniqueKey: 'id',
			onCheck: (rows) => {
				console.log('checked rows', rows);
			},
			onRowClick: (datum) => {
				console.log(datum);
			},
			headerData: [
				{ title: 'Name', id: 'name', size: 'lg' },
				{ title: 'Age', id: 'age', size: 'sm' },
				{ title: 'Designation', id: 'designation' },
				{ title: 'State', id: 'state', size: 'sm' },
			],
			expandable: ({ datum }) => {
				return <Text variant='h2'>Welcome {datum.name}!</Text>;
			},
			customCells: {
				body: () => {
					return {
						name: ({ datum, expandableProps, ...rest }) => {
							return (
								<TableCellV2
									{...rest}
									component3={
										<Button
											{...expandableProps}
											title={
												expandableProps?._expanded ? 'Collapse' : 'Expand'
											}
											size='auto'
										/>
									}
								/>
							);
						},
					};
				},
			},
		}}>
		{Template.bind({})}
	</Story>
</Canvas>
