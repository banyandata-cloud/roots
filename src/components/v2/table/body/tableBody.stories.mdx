export const Template = (args) => {
	const [checkedRows, setCheckedRows] = useState([]);
	return <TableBody {...args} checkedRows={checkedRows} setCheckedRows={setCheckedRows} />;
};

# Table Body

> Container for table body rows.

## Usage

```js
import { TableBody } from '@banyan_cloud/roots';
```
